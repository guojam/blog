# 使用 ngComponentOutlet 动态排序组件

import { Component, OnInit, Inject, Injector } from '@angular/core';

/** 子组件注入器token管理 */
import { InjectionToken } from '@angular/core';
export const TOKEN_SUBCOMP_A = new InjectionToken<any>('SubCompA_token');
export const TOKEN_SUBCOMP_B = new InjectionToken<any>('SubCompB_token');
export const TOKEN_SUBCOMP_C = new InjectionToken<any>('SubCompC_token');

/** 子组件定义 */
@Component({
    template: '<div>{{ data.title }}</div>',
})
class SubAComponent implements OnInit {
    data: any;
    constructor(@Inject(TOKEN_SUBCOMP_A) private dataInjector: any) {}
    ngOnInit() {
        this.data = this.dataInjector;
    }
}
// 同样定义 SubBComponent/SubCComponent ...

/** 组件动态排序及数据注入 */
@Component({
    template: `
        <div class="order-list">
            <ng-container *ngFor="let item of sortedComponent">
                <ng-container
                    *ngComponentOutlet="item.component; injector: item.injector"
                ></ng-container>
            </ng-container>
        </div>
    `,
})
export class OrderCompComponent implements OnInit {
    sortedComponent = <any>[];

    sortedMark: {
        [key: string]: any;
    } = {};

    constructor(private injector: Injector) {}

    ngOnInit() {
        this.initData();

        // 修改注入数据
        setTimeout(() => {
            this.sortedMark['subCompA']['injector'] = this.setInjector(
                TOKEN_SUBCOMP_A,
                {
                    title: '新标题',
                }
            );
        }, 3000);
    }

    /** 初始化动态组件数据 */
    private async initData() {
        // 异步获取已排序的组件信息
        const compList = await [
            {
                code: 'subCompB',
                data: {
                    title: '子组件B',
                },
            },
            {
                code: 'subCompC',
                data: {
                    title: '子组件C',
                },
            },
            {
                code: 'subCompA',
                data: {
                    title: '子组件A',
                },
            },
        ];
        compList.map((item) => {
            let component, token;
            switch (item['code']) {
                case 'subCompA':
                    component = SubAComponent;
                    token = TOKEN_SUBCOMP_A;
                    break;
                case 'subCompB':
                    component = SubBComponent;
                    token = TOKEN_SUBCOMP_B;
                    break;
                case 'subCompC':
                    component = SubCComponent;
                    token = TOKEN_SUBCOMP_C;
                    break;
                default:
                    break;
            }
            item['component'] = component;
            // 生成子组件注入器
            item['injector'] = this.setInjector(token, item['data']);
            this.sortedMark[item['code']] = item;
        });
        this.sortedComponent = compList;
    }

    /** 设置动态组件注入器 */
    private setInjector(token: any, data: any) {
        return Injector.create({
            providers: [{ provide: token, useValue: data }],
            parent: this.injector,
        });
    }
}
